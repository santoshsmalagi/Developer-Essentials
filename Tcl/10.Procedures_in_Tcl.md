# Procedures in Tcl

```Tcl
# simple Tcl procedure to add two numbers
proc add {a b} {
    return [expr $a + $b]
}

proc seperator {} {
    puts "========================================"
}

set sum [ add 5 8 ]
seperator
puts "5 + 8 = $sum"
seperator;

# proc to calculate factorial
proc factorial {val} {
    set result 1
    while {$val>0} {
        set result [expr $result*$val]
        incr val -1
    }
return $result
}
```
